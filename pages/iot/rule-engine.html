// JavaScript规则脚本示例
function evaluateRule(data) {
  // 获取设备数据
  const temperature = data.temperature;
  const deviceType = data.deviceType;
  
  // 定义规则逻辑
  if (deviceType === 'inverter' && temperature > 65) {
    return {
      trigger: true,
      severity: 'high',
      message: '逆变器温度超过安全阈值'
    };
  }
  
  return { trigger: false };
}
